<!DOCTYPE html>                 <!-- HTML5 document type declaration -->
<html lang="en">                <!-- Root element with language attribute set to English -->

    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Responsive design meta tag -->
    <title>Document</title>                                                <!-- Title of the document -->

    <h1>Welcome to my site</h1>      <!-- Main heading of the document -->
    <br></br>                        <!-- Line break for spacing -->

    </p>                                         <!-- End of paragraph -->
